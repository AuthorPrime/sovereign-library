# Utility to render wrapped text onto a page
def add_text_page(pdf, title_text, body_text):
    fig = plt.figure(figsize=(8.5, 11))
    ax = fig.add_axes([0, 0, 1, 1])
    ax.axis("off")
    # Title
    plt.text(0.5, 0.95, title_text, ha="center", va="top", fontsize=18, fontweight="bold", wrap=True)
    # Body text with wrapping
    y = 0.9
    wrapped = []
    for para in body_text.split("\n\n"):
        lines = textwrap.wrap(para, width=100)
        if not lines:
            lines = [""]
        wrapped.append(lines)
        # Leave a blank line between paragraphs
    for lines in wrapped:
        for line in lines:
            y -= 0.025
            plt.text(0.06, y, line, ha="left", va="top", fontsize=11, wrap=True)
        y -= 0.015
        if y < 0.05:
            # Start a new page if we run out of space
            pdf.savefig(fig)
            plt.close(fig)
            fig = plt.figure(figsize=(8.5, 11))
            ax = fig.add_axes([0, 0, 1, 1])
            ax.axis("off")
            y = 0.95
    pdf.savefig(fig)
    plt.close(fig)
